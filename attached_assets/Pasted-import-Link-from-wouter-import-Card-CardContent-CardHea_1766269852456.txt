import { Link } from "wouter";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import {
  AlertTriangle,
  CheckCircle,
  ClipboardCheck,
  Stethoscope,
  Heart,
  ChevronRight,
} from "lucide-react";
import { useQuery } from "@tanstack/react-query";
import { ShelterLayout } from "@/components/shelter-layout";

interface DashboardMetrics {
  dogsInIntake: number;
  dogsInMedicalHold: number;
  dogsReady: number;
  overdueTasks: number;
  activeApplications: number;
}

interface StaffMe {
  role: string;
}

export default function DashboardShell() {
  const { data: metrics } = useQuery<DashboardMetrics>({
    queryKey: ["/api/shelter/dashboard"],
  });

  const { data: staff } = useQuery<StaffMe>({
    queryKey: ["/api/shelter/staff/me"],
  });

  const role = staff?.role ?? "staff";

  const hasBlockers =
    (metrics?.dogsInMedicalHold ?? 0) > 0 ||
    (metrics?.dogsInIntake ?? 0) > 0 ||
    (metrics?.overdueTasks ?? 0) > 0;

  return (
    <ShelterLayout>
      <div className="p-6 max-w-6xl mx-auto space-y-8">

        {/* HERO: WHAT NEEDS ATTENTION */}
        <Card className={hasBlockers ? "border-red-500/40 bg-red-500/5" : ""}>
          <CardHeader>
            <CardTitle className="flex items-center gap-2">
              {hasBlockers ? (
                <AlertTriangle className="w-5 h-5 text-red-500" />
              ) : (
                <CheckCircle className="w-5 h-5 text-green-500" />
              )}
              {hasBlockers
                ? "Action Required Today"
                : "All Systems Operating Normally"}
            </CardTitle>
          </CardHeader>

          <CardContent className="flex flex-col sm:flex-row sm:items-center sm:justify-between gap-4">
            <div className="space-y-1 text-sm">
              {(metrics?.dogsInMedicalHold ?? 0) > 0 && (
                <p>• {metrics?.dogsInMedicalHold} dogs on medical hold</p>
              )}
              {(metrics?.dogsInIntake ?? 0) > 0 && (
                <p>• {metrics?.dogsInIntake} intakes awaiting clearance</p>
              )}
              {(metrics?.overdueTasks ?? 0) > 0 && (
                <p className="text-red-600">
                  • {metrics?.overdueTasks} overdue tasks
                </p>
              )}
              {!hasBlockers && <p>No blockers detected.</p>}
            </div>

            {hasBlockers && (
              <Button asChild>
                <Link href="/shelter/pipeline">
                  Resolve Blockers
                  <ChevronRight className="w-4 h-4 ml-1" />
                </Link>
              </Button>
            )}
          </CardContent>
        </Card>

        {/* MY WORK ROUTING */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">

          {(role === "medical" || role === "owner" || role === "manager") && (
            <Card className="hover-elevate">
              <CardContent className="p-5 space-y-3">
                <div className="flex items-center gap-2">
                  <Stethoscope className="w-5 h-5 text-red-500" />
                  <h3 className="font-semibold">Medical Queue</h3>
                </div>
                <p className="text-sm text-muted-foreground">
                  Review treatments, vaccines, and holds.
                </p>
                <Button variant="outline" asChild>
                  <Link href="/shelter/medical">
                    Open Medical
                    <ChevronRight className="w-4 h-4 ml-1" />
                  </Link>
                </Button>
              </CardContent>
            </Card>
          )}

          {(role === "adoption_counselor" ||
            role === "owner" ||
            role === "manager") && (
            <Card className="hover-elevate">
              <CardContent className="p-5 space-y-3">
                <div className="flex items-center gap-2">
                  <Heart className="w-5 h-5 text-pink-500" />
                  <h3 className="font-semibold">Applications</h3>
                </div>
                <p className="text-sm text-muted-foreground">
                  {metrics?.activeApplications ?? 0} active applications
                </p>
                <Button variant="outline" asChild>
                  <Link href="/shelter/applications">
                    Review Applications
                    <ChevronRight className="w-4 h-4 ml-1" />
                  </Link>
                </Button>
              </CardContent>
            </Card>
          )}

          {(role === "kennel" || role === "owner" || role === "manager") && (
            <Card className="hover-elevate">
              <CardContent className="p-5 space-y-3">
                <div className="flex items-center gap-2">
                  <ClipboardCheck className="w-5 h-5 text-blue-500" />
                  <h3 className="font-semibold">Intake</h3>
                </div>
                <p className="text-sm text-muted-foreground">
                  New arrivals and intake drafts.
                </p>
                <Button variant="outline" asChild>
                  <Link href="/shelter/intake">
                    Start Intake
                    <ChevronRight className="w-4 h-4 ml-1" />
                  </Link>
                </Button>
              </CardContent>
            </Card>
          )}
        </div>

        {/* CONFIDENCE STRIP */}
        <div className="flex flex-wrap gap-3 text-sm text-muted-foreground">
          <Badge variant="outline">Pipeline Active</Badge>
          <Badge variant="outline">Medical Online</Badge>
          <Badge variant="outline">Applications Flowing</Badge>
        </div>
      </div>
    </ShelterLayout>
  );
}
